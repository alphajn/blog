package.json 配置文件 #

{
  "name": "@fe/utils",
  "type": "module",
  "version": "0.0.1",
  "description": "前端工具集合",
  "author": "alpha.wu",
  "license": "BSD-3-Clause",
  "keywords": [
    "utils"
  ],
  "exports": {
    ".": {
      "types": "./dist/index.d.ts",
      "import": "./dist/index.mjs",
      "default": "./dist/index.mjs"
    },
    // 如果是分包打包需要配置
    "./bar": {
      "types": "./dist/bar.d.ts",
      "import": "./dist/bar.mjs",
      "default": "./dist/bar.mjs"
    },
    "./*": "./*"
  },
  "main": "./dist/index.mjs",
  "module": "./dist/index.mjs",
  "types": "./dist/index.d.ts",
  "files": [
    "dist",
    "*.md",
    "*.d.ts"
  ],
  "peerDependenciesMeta": {},
  "dependencies": {},
  "devDependencies": {}
}

注意

如果是分类包并且没有统一打包入口，需要配置 exports 对应的模块文件.

例如：

{
  "exports": {
    "./bar": {
      "types": "./dist/bar.d.ts",
      "import": "./dist/bar.mjs",
      "default": "./dist/bar.mjs"
    }
  }
}
