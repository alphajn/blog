深度图 ​

Demo ​

Install ​

bash
# 使用 npm
npm i fejn-depth

# 使用 yarn
yarn add fejn-depth

# 使用 pnpm
pnpm add fejn-depth

使用 ​

vue
<script setup lang="ts">
import { onMounted, reactive } from 'vue';
import { Depth } from 'fejn-depth';

const state = reactive({
    el: null,
    chart: null,
    buys: [
        [200000.69, 1000.0004],
        [300000.72, 2000.0006],
        [400000.3, 60.999],
    ],
    sells: [
        [200001.6329, 1000.0005],
        [300001.72, 200.0006],
        [400001.3, 60.999],
    ],
});

onMounted(() => {
    state.chart = new Depth(state.el, {
        theme: 'light',
    });
    state.chart?.setData(state.buys, state.sells);
});
</script>

<template>
    <div :ref="(el) => state.el = el" />
</template>

el ​

初始化 html 元素

初始化配置选项

样式 ​

ts
{
    grid: {

        color: 'black', // 文本颜色

实例函数 ​

setData() ​

设置图表数据。

  • 类型

    ts
    type ArrayList = (string | number)[]

/**
 * 设置数据并触发图表更新
 * @param buy 买数据列表，每个元素是一个包含价格和数量的数组
 * @param sell 卖数据列表，格式同买数据列表
 */
function setData(buy: ArrayList[], sell: ArrayList[]): void;

  • 示例

    ts
     const buys = [
    [200000.69, 1000.0004],
    [300000.72, 2000.0006],
    [400000.3, 60.999],
];
const sells = [
    [200001.6329, 1000.0005],
    [300001.72, 200.0006],
    [400001.3, 60.999],
];
setData(buys, sells);

函数 ​

formatList() ​

格式化数据。

  • 类型

    ts
    type ArrayList = (string | number)[];
type ListItem = {
    price: string
    quantity: string
    total: string
}

/**
 * 将输入的数组转换为格式化的列表
 * @param arr 输入数组，每个元素是一个包含价格和数量的数组
 * @returns 返回一个列表，每个元素包含价格、数量和累计总量
 */
function formatList(arr: ArrayList[]): ListItem[]

  • 示例

    ts
    const buys = [
    [100.1, 1000.0004],
    [100.2, 2000.0006],
    [100.3, 60.666],
];

const list = formatList(buys);
list = [
    {
        "price": "100.1",
        "quantity": "1000.0004",
        "total": "1000.0004"
    },
    {
        "price": "100.2",
        "quantity": "2000.0006",
        "total": "3000.001"
    },
    {
        "price": "100.3",
        "quantity": "60.666",
        "total": "3060.667"
    }
];

drawLine() ​

在canvas上绘制一条线。

  • 类型

    ts
    /**
 * 在canvas上绘制一条线
 * @param ctx canvas的2D绘图上下文
 * @param w 线的宽度
 * @param color 线的颜色
 * @param from 线的起点坐标，格式为[x, y]
 * @param to 线的终点坐标，格式为[x, y]
 * @param dashed 指定线是否为虚线，默认为false
 */
function drawLine (ctx: CanvasRenderingContext2D, w: number, color: string, from: number[], to: number[], dashed = false): void

drawArc() ​

在canvas上绘制一个圆形。

  • 类型

    ts
    /**
 * 在canvas上绘制一个圆形
 * @param ctx canvas的2D绘图上下文
 * @param x 圆心的x坐标
 * @param y 圆心的y坐标
 * @param color 圆的填充颜色
 * @param radius 圆的半径，默认为8
 */
function drawArc (ctx: CanvasRenderingContext2D, x: number, y: number, color: string, radius = 8): void

drawRect() ​

在canvas上绘制一个矩形，支持圆角和阴影效果。

  • 类型

    ts
    /**
 * 在canvas上绘制一个矩形，支持圆角和阴影效果
 * @param ctx canvas的2D绘图上下文
 * @param bg 矩形的背景色
 * @param x 矩形左上角的x坐标
 * @param y 矩形左上角的y坐标
 * @param w 矩形的宽度
 * @param h 矩形的高度
 * @param shadowBlur 阴影的模糊度，默认为20
 * @param radius 矩形边角的圆角半径，默认为8
 */
function drawRect (ctx: CanvasRenderingContext2D, bg: string, x: number, y: number, w: number, h: number, shadowBlur = 20, radius = 8): void

getTextWidth() ​

获取指定文本的宽度。

  • 类型

    /**

 * @returns 返回文本的宽度，如果无法测量则返回0
 */
function getTextWidth (ctx: CanvasRenderingContext2D, text: string): number;
