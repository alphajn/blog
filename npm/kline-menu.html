import { Menu } from 'fejn-kline/vue';

const state = reactive({
    theme: 'light',
    lang: 'zh-CN',

    graph: 1,

    loading: false, // 数据加载状态
});

const methods = {
    // 选择时间间隔的处理函数
    selectIntervalHandle(interval: string) {
        state.interval = interval;
    },
    // 编辑时间间隔的处理函数
    editIntervalHandle(intervals: config.Interval[]) {
        state.show_intervals = intervals;
    },
    // 截图的处理函数
    screenshotHandle() {
        // TODO::
    },
    // 全屏切换的处理函数
    fullScreenHandle() {
        state.fullscreen = !state.fullscreen;
    },
    // 选择图表类型的处理函数
    selectGraphHandle(graphValue: number) {
        state.graph = graphValue;
    },
};
</script>

<template>
    <Menu
        type="simple"
        :theme="state.theme"
        :lang="state.lang"

        :intervals="state.intervals"

        @edit-interval="methods.editIntervalHandle"
        @screenshot="methods.screenshotHandle"
        @full-screen="methods.fullScreenHandle"
        @select-graph="methods.selectGraphHandle"
    />
</template>

使用 ​

点我查看使用代码
vue
<script lang="ts" setup>
// 导入所需模块和类型
import { Menu } from 'fejn-kline/vue';
import type { config } from 'fejn-kline/utils';

// 定义组件的响应式状态
const state = reactive({
    theme: 'light',
    lang: 'zh-CN',
    interval: '60', // 默认时间间隔
    intervals:[ '1', '3', '5', '15', '30', '60', '120', '240', '360', '720', '1D', '1W', '1M' ], // 支持的时间选择间隔
    show_intervals: ['1', '15', '60', '240', '1D', '1W'], // 默认展示的时间选择间隔
    graph: 1,
    fullscreen: false // 当前全屏状态
    loading: false, // 数据加载状态
});

// 定义组件的方法
const methods = {
    // 选择时间间隔的处理函数
    selectIntervalHandle(interval: string) {
        state.interval = interval; // 更新状态中的时间间隔
    },
    // 编辑时间间隔的处理函数
    editIntervalHandle(intervals: config.Interval[]) {
        state.show_intervals = intervals; // 更新显示的时间间隔
    },
    // 显示指标的处理函数
    showIndicatorHandle() {
        // TODO::
    },
    // 设置的处理函数
    setHandle() {
        // TODO::
    },
    // 截图的处理函数
    screenshotHandle() {
        // TODO::
    },
    // 全屏切换的处理函数
    fullScreenHandle() {
        state.fullscreen = !state.fullscreen; // 切换全屏状态
    },
    // 选择图表类型的处理函数
    selectGraphHandle(graphValue: number) {
        state.graph = graphValue; // 更新图表类型
    },
};
</script>

<template>
    <!-- K线图菜单组件 -->
    <Menu
        type="tv"
        :theme="state.theme"
        :lang="state.lang"
        :interval="state.interval"
        :intervals="state.intervals"
        :selected-intervals="state.show_intervals"
        :graph="state.graph"
        :fullscreen="state.fullscreen"
        :loading="state.loading"
        @select-interval="methods.selectIntervalHandle"
        @edit-interval="methods.editIntervalHandle"
        @show-indicator="methods.showIndicatorHandle"

API ​

属性 ​

属性名说明类型默认值
typechart类型simple | tvsimple
theme主题drak | lightdark
lang多语言Langen-US
interval当前选中的时间区间Interval60
intervals支持的时间区间列表Interval[]['1', '3', '5', '15', '30', '60', '120', '240', '360', '720', '1D', '3D', '1W', '1M']
selected-intervals当前展示时间区间列表Interval[]['1', '15', '60', '240', '1D', '1W']
graph当前图表选择类型number1
fullscreen是否全屏booleanfalse
loading数据获取中booleanfalse

Lang 类型 ​

ts
type Lang = 'en-US' | 'zh-CN' | 'zh-TW' | 'ja-JP' | 'vi-VN' | 'es-ES' | 'ru-RU' | 'tr-TR' | 'pt-BR' | 'fr-FR' | 'th-TH' | 'id-ID' | 'ko-KR';

Interval 类型 ​

ts
type Interval = '1' | '3' | '5' | '15' | '30' | '60' | '120' | '240' | '360' | '720' | '1D' | '3D' | '1W' | '1M';
