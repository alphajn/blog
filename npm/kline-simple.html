Install ​

bash
# 使用 npm
npm i fejn-kline fejn-number klinecharts

# 使用 yarn
yarn add fejn-kline fejn-number klinecharts

# 使用 pnpm
pnpm add fejn-kline fejn-number klinecharts

使用 ​

点我查看使用代码
vue
<script setup lang="ts">
import { Simple } from 'fejn-kline/vue';
import type { config } from 'fejn-kline/utils';

const state = reactive({
    el: null as Ref | null,
    chartEl: null as Ref | null, // tv实例
    theme: 'light',
    lang: 'zh-CN',
    graph: 1,
    loading: false, // 数据加载状态
    list: [] as config.KLineData[],
    latestData: null as config.KLineData | null, // 最新数据
});

const methods = {
    loadMore(ts: number | null, cb: (list: config.KLineData[]) => void) {
        // ::TODO
    },
}
</script>

<template>
<div :ref="el => state.el = el">
    <Simple
        :ref="el => state.chartEl = el"
        watermark="Alpha"
        :theme="state.theme"
        :lang="state.lang"
        :graph="state.graph"
        :price-precision="4"
        :quantity-precision="4"
        :loading="state.loading"
        loading-color="red"
        :list="state.list"
        :latest-data="state.latest_data"
        :load-more="methods.loadMore"
    />
</div>
</template>

Theme类型 ​

ts
type Theme = 'dark' | 'light';

Lang类型 ​

ts
type Lang = 'en-US' | 'zh-CN' | 'zh-TW' | 'ja-JP' | 'vi-VN' | 'es-ES' | 'ru-RU' | 'tr-TR' | 'pt-BR' | 'fr-FR' | 'th-TH' | 'id-ID' | 'ko-KR';

KlineData类型 ​

ts
type KlineData = {
    timestamp: number; // 时间戳，毫秒级别，必要字段
    open: number; // 开盘价，必要字段
    high: number; // 最高价，必要字段
    low: number; // 最低价，必要字段
    close: number; // 收盘价，必要字段
    volume?: number; // 成交量，非必须字段
    turnover?: number // 成交额
}

Chart类型 ​

ts
type Chart = KlineChart & {
    $updateTooltip: () => void // 更新tooltip
    $setTheme: (theme: `${Theme}`) => void // 更新主题

    $setGraph: (shape: number) => void // 设置图表形状
    $addDefaultMa: () => void // 手动添加ma指标
    $addVol: () => void // 手动添加成交量
} | null;
