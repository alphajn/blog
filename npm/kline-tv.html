Install ​

bash
# 使用 npm
npm i fejn-kline

# 使用 yarn
yarn add fejn-kline

# 使用 pnpm
pnpm add fejn-kline

使用 ​

注意

  1. 如果使用 TradingView 需要把对应版本的 charting_library 放到静态文件目录下，并设置 libraryPath 属性为 charting_library 路径
  2. 如果需要自定义样式需要把对应 css 文件放到 charting_library 目录下，并设置 custom_css_url 属性为文件名
点我查看使用代码
vue
<script setup lang="ts">
import { Menu, Tv } from 'fejn-kline/vue';

// 定义组件的响应式状态
const state = reactive({
    el: null as Ref | null,
    chartEl: null as Ref | null, // tv实例
    theme: 'light',
    lang: 'zh-CN',
    interval: '60', // 默认时间间隔
    intervals:[ '1', '3', '5', '15', '30', '60', '120', '240', '360', '720', '1D', '1W', '1M' ], // 支持的时间选择间隔
    show_intervals: ['1', '15', '60', '240', '1D', '1W'], // 默认展示的时间选择间隔
    graph: 1,
    fullscreen: false // 当前全屏状态
    loading: false, // 数据加载状态
    list: [] as config.TvKlineData[],
    latestData: null as config.TvKlineData | null, // 最新数据
});

const methods = {
    // 选择时间间隔的处理函数
    selectIntervalHandle(interval: string) {
        state.interval = interval; // 更新状态中的时间间隔
    },
    // 编辑时间间隔的处理函数
    editIntervalHandle(intervals: config.Interval[]) {
        state.show_intervals = intervals; // 更新显示的时间间隔
    },
    // 显示指标的处理函数
    showIndicatorHandle() {
        const activeChart = state.chartEl?.chart?.activeChart();
        activeChart?.executeActionById('insertIndicator');
    },
    // 设置的处理函数
    setHandle() {
        const activeChart = chartEl?.chart?.activeChart();
        activeChart?.executeActionById('chartProperties');
    },
    // 截图的处理函数
    screenshotHandle() {
        state.chartEl?.chart?.takeScreenshot();
    },
    // 全屏切换的处理函数
    fullScreenHandle() {
        if (state.fullscreen && screen.exitFullscreen) {
            screen.exitFullscreen();
        } else {
            screen.requestFullscreen(state.el);
        }
    },
    // 选择图表类型的处理函数
    selectGraphHandle(graphValue: number) {
        state.graph = graphValue; // 更新图表类型
    },
    /** 加载更多数据 */
    loadMore(ts: number | null, cb: (list: config.TvKlineData[]) => void) {
        // TODO::
    },
    plusClickHandle(params: PlusClickParams) {
        // TODO::
    }
}
</script>

<template>
<div
    :ref="el => state.el = el"
    class="rd-6px b-1px b-solid h-400px flex flex-col bg-[var(--vp-c-bg)] overflow-hidden"
>
    <Menu
        type="tv"
        :theme="state.theme"
        :lang="state.lang"
        :interval="state.interval"
        :intervals="state.intervals"
        :selected-intervals="state.show_intervals"
        :graph="state.graph"
        :fullscreen="state.fullscreen"
        :loading="state.loading"
        @select-interval="methods.selectIntervalHandle"
        @edit-interval="methods.editIntervalHandle"
        @show-indicator="method.showIndicatorHandle"
        @set="methods.setHandle"
        @screenshot="methods.screenshotHandle"
        @full-screen="methods.fullScreenHandle"
        @select-graph="methods.selectGraphHandle"
    />
    <Tv
        :ref="el => state.chartEl = el"
        :library-path="libraryPath"
        store-key="tv"
        exchange="exchange"
        watermark="Alpha"
        :theme="state.theme"
        :lang="state.lang"
        symbol="BTCUSDT"
        :interval="state.interval"
        :graph="state.graph"
        :price-precision="8"
        :quantity-precision="8"
        :loading="state.loading"
        loading-color="green"
        :limit-size="1000"
        :list="state.list"
        :latest-data="state.latestData"
        :load-more="state.loadMore"
        @plus-click="methods.plusClickHandle"
    />
</div>
</template>

Theme类型 ​

ts
type Theme = 'dark' | 'light';

Lang类型 ​

ts
type Lang = 'en-US' | 'zh-CN' | 'zh-TW' | 'ja-JP' | 'vi-VN' | 'es-ES' | 'ru-RU' | 'tr-TR' | 'pt-BR' | 'fr-FR' | 'th-TH' | 'id-ID' | 'ko-KR';

Interval类型 ​

ts
type Interval = '1' | '3' | '5' | '15' | '30' | '60' | '120' | '240' | '360' | '720' | '1D' | '3D' | '1W' | '1M';

TvKlineData类型 ​

ts
type TvKlineData = {
    id: string; // 必须字段，唯一id，例如：`${symbol}_${interval}`
    time: number; // 时间戳，毫秒级别，必要字段
    open: number; // 开盘价，必要字段
    high: number; // 最高价，必要字段
    low: number; // 最低价，必要字段
    close: number; // 收盘价，必要字段
    volume?: number; // 成交量，非必须字段
}
