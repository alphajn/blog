编辑器统一风格 editorConfig ​

使用说明 ​

项目根目录添加 .editorconfig 文件

sh
# 告诉EditorConfig插件，这是根文件，不用继续往上查找
root = true

# 匹配.js结尾的文件
[*.js]

# 匹配.js .scss .css 结尾的文件
[*.{js,scss,css}]

# 设置字符集
charset = utf-8

# 缩进风格，可选"space"、"tab"
indent_style = space

# 缩进空格数
indent_size = 4

# 结尾换行符，可选"lf"、"cr"、"crlf"
end_of_line = lf

# 删除一行中的前后空格
trim_trailing_whitespace = true

# 在文件结尾插入新行
insert_final_newline = true

# 每行最多字符数
max_line_length = 150
