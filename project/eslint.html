配置 eslint ​

  1. 使用封装扩展 fejn-eslint-vue
  2. 详细文档 @antfu/eslint-config
  3. Eslint文档

安装 ​

sh
pnpm add -D eslint fejn-eslint-vue

添加配置 eslint.config.js文件 ​

ts
const eslint = require('fejn-eslint-vue');

module.exports = eslint(
    {
        ignores: [
            '**/charting_library.js',
            '**/charting_library/*',
        ],
    },
    {
        // flat configs
    }
);

添加命令到 package.json ​

json
{
    "scripts": {
        "lint": "eslint .",
        "lint:fix": "eslint . --fix",
        "lint:viewer": "npx eslint-flat-config-viewer"
    }
}

配置 VS Code 自动修复 ​

.vscode/settings.json 添加以下内容

json
{
    // Disable the default formatter, use eslint instead
    "prettier.enable": false,
    "editor.formatOnSave": false, // 保存时使用VSCode 自身格式化程序格式化
    // Auto fix
    "editor.codeActionsOnSave": {
        "source.fixAll.eslint": "explicit",
        "source.organizeImports": "never"
    },
    "eslint.validate": [
        "javascript",
        "javascriptreact",
        "typescript",
        "typescriptreact",
        "vue",
        "html",
        "markdown",
        "json",
        "jsonc",
        "yaml",
        "toml",
        "xml",
        "gql",
        "graphql",
        "astro",
        "svelte",
        "css",
        "less",
        "scss",
        "pcss",
        "postcss"
    ],
    "editor.rulers": [
        150
    ],
    "scss.lint.unknownAtRules": "ignore", // scss @规则忽略
    "editor.quickSuggestions": {
        "other": "on",
        "comments": "on",
        "strings": "on"
    }
}
