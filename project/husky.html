git 钩子 husky ​

文档 husky

其他 git hooks ​

  1. simple-git-hooks

安装 ​

sh
pnpm add -D husky lint-staged

自动启用 Git 挂钩 ​

sh
npm pkg set scripts.prepare="husky"

# 或

# 手动添加到package.json文件
{
  "scripts": {
    "prepare": "husky"
  }
}

# 执行
pnpm i

添pre-commit钩子 ​

sh
# 1. 添加文件
.husky/pre-commit

# 2. 添加以下内容

#!/usr/bin/env sh

pnpm lint-staged

添加执行命令 ​

json
// package.json 里添加以下内容
{
    "lint-staged": {
        "*.{js,jsx,ts,tsx,vue}": [
            "pnpm lint:fix"
        ]
    }
}
