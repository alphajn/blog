git 常用命令 ​

官网

配置 ​

生成密钥 SSH key ​

bash
ssh-keygen -t rsa -C "email@example.com"

设置/查看配置 ​

bash
# 查看用户名
git config user.name
# 查看全局用户名
git config --global user.name
# 设置用户名
git config user.name 'alpha.wu'
# 设置全局用户名
git config --global user.name 'alpha.wu'

# 查看邮箱
git config user.email
# 查看全局邮箱
git config --global user.email
# 设置邮箱
git config user.email 'alpha.wu@test.com'
# 设置全局邮箱
git config --global user.email 'alpha.wu@test.com'

tag 命令 ​

删除 ​

批量删除本地 tag ​

bash
git tag -d $(git tag -l)

批量删除远程 tag ​

bash
git push origin --delete $(git tag -l)

remote 命令 ​

查看仓库 ​

bash
git remote

添加仓库地址 ​

bash
git remote add origin git@github.com:alphajn/fe-monorepo.git

修改仓库地址 ​

bash
git remote set-url origin git@github.com:alphajn/fe-monorepo.git

删除仓库地址 ​

bash
git remote rm git@github.com:alphajn/fe-monorepo.git

刷新仓库分支 ​

bash
git remote update origin --prune

查看仓库信息 ​

bash
git remote -v

查看仓库状态 ​

bash
git remote show origin

commit 提交 ​

添加描述提交已git add的内容 ​

bash
git commit -m 'feat: 提交描述'

自动 add 并提交 ​

注意

新添加文件不生效需要先git add .

bash
git commit -am 'feat: 提交描述'

合并提交/修改commit信息 ​

把暂存区的内容(git add .)合并到当前git commit还未git push生成新的commit, 如果已经git push合并后需要强制提交

bash
git add .
git commit --amend

# 修改 commit 信息
git commit --amend

push 推送 ​

推送已 commit 的内容 ​

bash
git push

推送并设置远程源 ​

bash
# git push -u [远程源] [对应分支]
git push -u origin master

强制推送 ​

bash
git push -f

更新代码 ​

fetch ​

更新源代码，需要手动合并到当前分支

bash
git fetch
# 或
git fetch --all

pull ​

bash
git pull # 更新合并代码
# 等于
git fetch
git merge

rebase 方式更新 ​

bash
git pull -r
# 或
git pull --rebase
# 更新origin代码到当前分支（变基）
git rebase origin

branch 分支 ​

创建分支 ​

bash
git branch <branch name>

创建并切换到当前分支 ​

bash
git checkout -b <branch name>

创建跟踪分支 ​

bash
git checkout -b <branch name> 远程源/<branch name>
# 或
git checkout --track 远程源/<branch name>

修改当前分支对应的远程分支 ​

bash
git branch -u <remote branch name>
# 或
git --set-upstream-to <remote branch name>

切换分支 ​

bash
git checkout <branch name>

查看本地分支 ​

bash
git branch

查看远程分支 ​

bash
git branch -r

查看所有分支 ​

bash
git branch -a

查看当前分支名 ​

bash
git rev-parse --abbrev-ref HEAD

查看当前分支提交 hash ​

bash
git rev-parse HEAD # 完整 hash
git rev-parse --short HEAD # 前7位简短 hash

查看分支提交信息 ​

bash
git branch -v

查看分支详细提交信息 ​

bash
git branch -vv

查看已合并的分支 ​

bash
git branch --merged

查看未合并的分支 ​

bash
git branch --no-merged

删除分支 ​

bash
git branch -d <branch name> <...> # 可以一次删除多个
# 强制删除指定分支
git branch -D <branch name>

删除远程分支 ​

bash
git push origin --delete <branch name>

清理分支 ​

bash
# 清理本地分支
git branch | xargs git branch -d
# 清理远程分支
git remote prune origin

log 命令 ​

基础查看 ​

bash
git log

单行显示提交信息 ​

bash
git log --oneline

图形化显示 ​

bash
git log --graph

查看diff ​

bash
git log -p

控制显示条数 ​

bash
# 显示最近4条提交记录
git log -4

显示提交统计信息 ​

bash
git log --stat

status ​

查看当前文件状态 ​

bash
git status

stash 贮存 ​

贮存 ​

bash
# 索引存入
git stash
# 或
git stash save 备注

查看 ​

bash
git stash list

取出贮存 ​

bash
# 只取出不会删除
git stash apply stash@\{0\}
# 取出并删除
git stash pop stash@\{0\}

删除 ​

bash
git stash drop

删除所有 ​

bash
git stash clear

恢复已删除 stash ​

bash
# 获取已删除列表
git fsck --lost-found
# 查看删除内容
git show [hash]
# 恢复
git merge [hash]

回滚/恢复修改文件 ​

放弃修改，还未提交到暂存区(未git add) ​

bash
# 放弃文件
git checkout <file name>
# 放弃当前工作区所有修改，不包含新增文件和文件夹
git checkout .
# 放弃工作区所有修改(包含新增文件和文件夹)
git clean -df

reset ​

从暂存区恢复到工作区 ​

bash
git reset <file> # 恢复指定文件
git reset . # 恢复所有文件

回滚已push指定版本代码 ​

回滚会连同提交记录一起修改，再次提交需要强制提交。

bash
# 回滚到 commit id 版本代码，当前版本之后修改的代码在工作区可以编辑
git reset <commit id>
# 回滚到 commit id 版本代码，放弃当前版本之后的修改
git reset --hard  <commit id>

revert 回滚 ​

回滚会生成一份新的记录，不会修改历史版本提交记录。

bash
git revert <commit id> # 回滚到指定版本
