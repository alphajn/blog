常用 TypeScript ​

基本类型 ​

  • 字符串: string

  • 数字: number

  • 真假值: boolean

  • 数组类型: number[] 或 Array<number> 或 [number, string]

  • 函数类型: Function

  • 任意类型: unknown、any

  • 不存在: null

  • 未初始化: undefined

  • 无返回值: void

  • 从不返回: never ，通常用于函数内部抛出错误:

    ts
    function fn() {
    throw new Error('error');
}

  • 函数: () => void 或 (arg: string) => void

用法 ​

函数用法 ​

ts
type Age = number | string; // 联合类型

// 函数参数类型
function greet(name: string, age?: Age) {
    // TODO::
}
// 函数返回值类型
function greet(name: string): string {
    return `hello ${name}`;
}
// 其余参数
function sum(n1: number, ...args: number[]) {
    return args.reduce((pre, cur) => pre += cur, n1);
}

对象用法 ​

ts
type Person = {
    name: string
    arg?: number
    fn(): void
    readonly c: undefined // 只读
    [index: number]: string // 索引签名
};

type Apple = Record<string, string>;
// 等同于
type Apple = {
    [propsName: string]: string
};

非空断言 ! ​

表示值不为 null 和 undefined

ts
function greet(age: number | null) {
    console.log(age!.toFixed());
}

function getArea(shape?: number) {
    return Math.PI * shape.radius! ** 2;
}

范类型 <> ​

ts
const arr: Array<number>;

type Greet = <T>(name: T) => T;

const greet: Greet<string> = name => name;

// 类型参数
type Box<T> = {
    name: T
};

let box: Box<string>;

// 修改对象类型
type Options2Boolean<T> = {
    [Property in keyof T]: boolean
};
type Options = {
    a: string
    b: number
};
type OptionsBoolean = Options2Boolean<Options>;
// OptionsBoolean = {
//     a: boolean
//     b: boolean
// }

类 ​

ts
class Point {
    x: number; // x = 0 有默认值

    readonly y: number; // x = 1
}

const pt = new Point();
pt.x = 0;
pt.y = 0;

模版类型 ​

ts
type ShortLang = 'tw' | 'hk';
type ShortLang1 = 'cn';

type Lang = `zh_${ShortLang}`;
// type Lang = 'zh_tw' | 'zh_hk'

type LangAll = `zh_${ShortLang | ShortLang1}`;
// type LangAll = 'zh_tw' | 'zh_hk' | 'zh_cn'

运算符 ​

typeof 类型运算符 ​

ts
type T = 'Hello world';
typeof T; // "string"

const map = {
    x: 'hello',
    y: 666,
};
type Map = typeof map;
// Map = {
//     x: string;
//     y: number;
// }

const arr = [
    { x: 'x', y: 1 },
    { x: 'x1', y: 2 },
];
type ArrItem = typeof arr[number];
// ArrItem = {
//     x: string,
//     age: number,
// };
type ArrItemX = typeof arr[number]['age']; // ArrItemX = string

索引访问类型 ​

ts
type Person = { age: number, name: string, alive: boolean };
type Age = Person['age']; // Age = number
type AgeName = Person['age' | 'name']; // AgeName = number | string
type PickPerson = Person[keyof Person]; // PickPerson = string | number | boolean

kyeof 运算符 ​

ts
type Person = {
    name: string
    age: number
};
type P = keyof Person; // P = "name" | "age"

type Map = { [k: string]: boolean };
type M = keyof Map; // M = string | number

合并类型 & ​

ts
type Box = {
    name: string
};

type Apple = {
    size: number
};

type BoxApple = Box & Size;
// BoxApple = {
//     name: string,
//     size: number,
// };

类型转换 ​

Awaited 等待 ​

这种类型旨在对 async await 或 Promise 上的 .then() 方法等操作进行建模 - 具体来说，是它们递归解开 Promises 的方式。

ts
type A = Awaited<Promise<string>>;
// type A = string

type B = Awaited<Promise<Promise<number>>>;
// type B = number

type C = Awaited<boolean | Promise<number>>;
// type C = number | boolean

Partial 可选 ​

类型全部设置成可选类型

ts
type A = {
    a: string
    b: string
};

type PartialA = Partial<A>;
// type A = {
//     a?: string,
//     b?: string,
// }

Required 必须 ​

类型全部设置必须类型

ts
type A = {
    a?: string
    b?: string
};

type B = Required<A>;
// type B = {
//     a: string
//     b: string
// }

Readonly 只读 ​

类型全部设置为只读

ts
type A = {
    a: string
    b: string
};

type B = Readonly<A>;
// type B = {
//     readonly a: string
//     readonly b: string
// }

Record 对象 ​

构造一个对象类型，其属性键为Keys，其属性值为Type。此实用工具可用于将一个类型的属性映射到另一个类型。

ts
type Name = 'a' | 'b' | 'c';
type A = Record<Name, string>;
// type A = {
//     a: string
//     b: string
//     c: string
// }

Pick 选取 ​

通过从 Type 中选取属性Keys（字符串文本或字符串文本的联合）来构造类型.

ts
type A = {
    a: string
    b: string
    c: number
};
type B = Pick<A, 'a' | 'b'>;
// type B = {
//     a: string
//     b: string
// }

Omit 排除 ​

通过从 Type 中选取所有属性，然后删除Keys（字符串文本或字符串文本的联合）来构造类型。Pick相反.

ts
type A = {
    a: string
    b: string
    c: number
};
type B = Omit<A, 'a' | 'b'>;
// type B = {
//     c: string
// }

Exclude 省略 ​

ts
type T0 = Exclude<'a' | 'b' | 'c', 'a'>;
// type T0 = 'b' | 'c'

type T1 = Exclude<'a' | 'b' | 'c', 'a' | 'b'>;
// type T1 = 'c'

type T2 = Exclude<string | number | (() => void), Function>;
// type T2 = string | number

type A = { a: 1, e: number } | { b: 1, f: number } | { c: 1, g: number };
type B = Exclude<A, { a: 1 }>;
// type B = { b: 1; f: number } | { c: 1; g: number }

Extract 提取 ​

跟 Exclude 相反

ts
type T0 = Extract<'a' | 'b' | 'c', 'a' | 'f'>;
// type T0 = 'a'

type T1 = Extract<string | number | (() => void), Function>;
// type T1 = () => void

type A = { a: 1, e: number } | { b: 1, f: number } | { c: 1, g: number };
type B = Extract<A, { a: 1 }>;
// type B = { a: 1; e: number }

NonNullable ​

通过从类型中排除 null 和 undefined 来构造Type.

ts
type T0 = NonNullable<string | number | undefined>;
// type T0 = string | number

type T1 = NonNullable<string[] | null | undefined>;
// type T1 = string[]

ReturnType 返回类型 ​

ts
type Fn = (x: string) => boolean;
type K = ReturnType<Fn>; // K = boolean

// 使用函数返回值类型
function f() {
    return { x: 10, y: 3 };
}
type P = ReturnType<typeof f>;
// P = {
//     x: number;
//     y: number;
// }

枚举 ​

数字枚举 ​

其中 Up 初始化为 1。从该点开始，以下所有成员都会自动递增。换句话说，Direction.Up 的值为 1，Down的值为 2，Left的值为 3，Right的值为 4

ts
enum Direction {
    Up = 1,
    Down,
    Left,
    Right,
}

字符串枚举 ​

ts
enum Direction {
    Up = 'UP',
    Down = 'DOWN',
    Left = 'LEFT',
    Right = 'RIGHT',
}
