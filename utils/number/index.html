plus #

  • 加法: plus(n1, n2)
  • 参数:
    • n1: string | number
    • n2: string | number
  • 返回值: string
  • 示例:
import { plus } from '@fe/utils/number';

plus('1', 2);            // '3'
plus('-1', 2);           // '1'
plus('0.1', 0.2);        // '0.3'
plus(0.1, 0.2);          // '0.3'
plus(0.7, '6.20000');    // '6.9'
plus('', 'abcd');        // '0'
plus('', '2');           // '0'

minus #

  • 减法： minus(n1, n2)
  • 参数:
    • n1: string | number
    • n2: string | number
  • 返回值: string

toFixed #

  • 小数位处理： toFixed(n1 [, dp[, rm]])
  • 参数:
    • n1: string | number
    • dp?: number | undefined 保留几位小数
    • rm?: 0 | 1 | 7 | undefined 保留小数最后一位处理方式，根据截取的后一位计算得出
      • 1 默认值，向下取整
      • 0 向上取整
      • 7 四舍五入
  • 返回值: string

    提示

    如果不传dp参数默认会舍弃掉小数后面的0。

    例如：toFixed('123.120000') 会返回 '123.12'

  • 示例:
import { toFixed } from '@fe/utils/number';

toFixed(1.12345);            // '1.12345'
toFixed(1.12300);            // '1.123' 没有截取长度会舍弃末尾的0
toFixed(1.12345, 4);         // '1.1234' 向下截断
toFixed(1.12345, 10);        // '1.1234500000' 自动补0
toFixed(1.12345, 2, 0);      // '1.13' 向上取整
toFixed(1.12345, 4, 7);      // '1.1235' 四舍五入
toFixed(1.12345, undefined); // '1.12345'

sd #

  • 整体精度： sd(n1 [, dp[, rm]])
  • 参数:
    • n1: string | number
    • dp?: number | undefined 截取精度
    • rm?: 0 | 1 | 7 | undefined 保留最后一位处理方式，根据截取的后一位计算得出
      • 1 默认值，向下取整
      • 0 向上取整
      • 7 四舍五入
  • 返回值: string
  • 示例:
import { sd } from '@fe/utils/number';

sd(12345.12345000, 2);       // '12000'
sd(12345.12345000, 4);       // '12340'
sd(12345.12345000, 3, 0);    // '12400' 向上取整
sd(12345.12345000, 4, 7);    // '12350' 四舍五入
sd(1.12345000, 2);           // '1.1'
sd(1.12345000, 3, 0);        // '1.13'  小数向上取整
sd(1.12345000, 4, 7);        // '1.123' 小数四舍五入
sd(1.12345000, 5, 7);        // '1.1235'

toFormat #

  • 格式化千分位&小数格式化： toFormat(n1 [, dp[, rm[, format]]])
  • 参数:
    • n1: string | number
    • dp?: number | undefined 保留几位小数
    • rm?: 0 | 1 | 7 | undefined 保留小数最后一位处理方式，根据截取的后一位计算得出
      • 1 默认值，向下取整
      • 0 向上取整
      • 7 四舍五入
    • format?: object 格式化
  • 返回值: string

    提示

    如果不传dp参数默认会舍弃掉小数后面的0。

    例如：toFormat('123.120000') 会返回 '123.12'

  • 示例:
import { toFormat } from '@fe/utils/number';

toFormat(123456789.12345);               // '123,456,789.12345'
toFormat(123456789.12300);               // '123,456,789.123' 舍弃末尾的0
toFormat('-123456789.12345');            // '-123,456,789.12345'
toFormat(123456789.12345, 4);            // '123,456,789.1234'
toFormat(123456789.12344, 4, 0);         // '123,456,789.1235' 向上取整
toFormat(123456789.12344, 4, 7);         // '123,456,789.1234' 四舍五入
toFormat(123456789.12345, 4, 7);         // '123,456,789.1235' 四舍五入
toFormat(123456789.12345, 10);           // '123,456,789.1234500000' 补零

// 格式化配置
const format = {
    prefix: '',                          // 前置字符串
    decimalSeparator: '.',               // 小数分隔符
    groupSeparator: ',',                 // 整数部分的分组分隔符
    groupSize: 3,                        // 整数部分的主要分组长度
    secondaryGroupSize: 0,               // 整数部分的二次分组大小
    fractionGroupSeparator: '',          // 小数部分的分组分隔符
    fractionGroupSize: 0,                // 小数部分的分组大小
    suffix: ' $',                        // 要附加的字符串
};

toFormat(123456789.12345, format);       // 123,456,789.12345 $' 格式化
toFormat(123456789.12345, 4, format);    // 123,456,789.1234 $'
toFormat(123456789.12345, 4, 0, format); // 123,456,789.1235 $'

toFormatUnit #

  • 格式化千分位带单位： toFormatUnit(n1 [, dp[, rm]])
  • 参数:
    • n1: string | number
    • dp?: number | undefined 保留几位小数
    • rm?: 0 | 1 | 7 | undefined 保留小数最后一位处理方式，根据截取的后一位计算得出
      • 1 默认值，向下取整
      • 0 向上取整
      • 7 四舍五入
import { toFormatUnit } from '@fe/utils/number';

toFormatUnit('1');                   // '1.00'       默认精度2位
toFormatUnit('999');                 // '999.00'
toFormatUnit('999', 0);              // '999'        精度0位
toFormatUnit('999', 4);              // '999.0000'   精度补0
toFormatUnit('1000.1');              // '1.00 K'     默认精度2位 千
toFormatUnit('1234.1', 4);           // '1.2341 K'
toFormatUnit('1234560.1234', 4);     // '1.2345 M'   4位精度 百万
toFormatUnit('123456000.1234');      // '123.45 M'
toFormatUnit('1234560000.1234', 4);  // '1.2345 B'   4位精度 10亿
toFormatUnit('1234.1234', 3, 0);     // '1.235 K'    最后一位向上取整
toFormatUnit('1234.1234', 3, 7);     // '1.234 K'    最后一位四舍五入
toFormatUnit('1234.5234', 3, 7);     // '1.235 K'

sum #

  • 求和： sum(n1 [, ...])
  • 参数:
    • n1: number | string
  • 返回值: string
  • 示例:
import { sum } from '@fe/utils/number';

sum('sadf');                        // '0'
sum('1', 2, 3, 4, 5, 6);            // '21'
sum('1', '1.123000', '2.877000')    // '5'
sum('1', '1.1234567', '2.8765433'); // '5'

random #

  • 返回值: string

import { random } from '@fe/utils/number';

random();    // '0.411793684723456789054678945678'
random(2);   // '0.41'
